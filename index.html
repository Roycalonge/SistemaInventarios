document.addEventListener("DOMContentLoaded", function() { const tabButtons = document.querySelectorAll(".tab-button"); const tabContents = document.querySelectorAll(".tab-content"); // Función para cargar datos desde un archivo JSON function loadData(tabId, url) { fetch(url) .then(response => response.json()) .then(data => { const contentDiv = document.getElementById(`${tabId}-content`); if (tabId === "categorias") { contentDiv.innerHTML = `
    ${data.map(item => `
  • ${item.nombre}
    • `).join("")}
`; } else if (tabId === "inventario-productos") { contentDiv.innerHTML = ` ${data.map(item => ` `).join("")}
Producto Cantidad Precio
${item.nombre} ${item.cantidad} $${item.precio.toFixed(2)}
`; } else if (tabId === "inventario-materia-prima") { contentDiv.innerHTML = `
    ${data.map(item => `
  • ${item.nombre} - ${item.cantidad} unidades
    • `).join("")}
`; } else if (tabId === "movimientos") { contentDiv.innerHTML = `
    ${data.map(item => `
  • ${item.tipo}: ${item.producto} - ${item.cantidad} unidades
    • `).join("")}
`; } }); } // Manejar pestañas tabButtons.forEach(button => { button.addEventListener("click", () => { tabButtons.forEach(btn => btn.classList.remove("active")); tabContents.forEach(content => content.classList.remove("active")); button.classList.add("active"); const tabId = button.getAttribute("data-tab"); // Cargar datos según la pestaña seleccionada if (tabId === "categorias") { loadData(tabId, "data/categorias.json"); } else if (tabId === "inventario-productos") { loadData(tabId, "data/inventario-productos.json"); } else if (tabId === "inventario-materia-prima") { loadData(tabId, "data/inventario-materia-prima.json"); } else if (tabId === "movimientos") { loadData(tabId, "data/movimientos.json"); } document.getElementById(tabId).classList.add("active"); }); }); // Cargar datos iniciales (primera pestaña) loadData("categorias", "data/categorias.json"); // Manejar el formulario document.getElementById("config-form").addEventListener("submit", function(event) { event.preventDefault(); alert("Configuración guardada correctamente."); }); });
